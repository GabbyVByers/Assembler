
label START
	LDI r0 #0
	STR r0 &snake_color
	LDI r0 #0
	STR r0 &head_x
	LDI r0 #0
	STR r0 &head_y

label MAIN_LOOP
	
	// snake_color++;
	// head_x++;
	// if (head_x == 256) {
	//     head_x = 0;
	//     head_y++;
	// }

	LDR r0 &snake_color
	ADI r0 #1
	STR r0 &snake_color
	LDR r0 &head_x
	ADI r0 #1
	LDI r1 #256
	CMP r0 r1
	LDI r0 #0 FLAG_EQUAL
	STR r0 &head_x
	LDR r0 &head_y FLAG_EQUAL
	ADI r0 #1 FLAG_EQUAL
	STR r0 &head_y FLAG_EQUAL


	JSR .DRAW_HEAD
	JMP .MAIN_LOOP

	
label DRAW_HEAD
	LDR r0 &snake_color
	LDR r2 &head_x
	LDR r3 &head_y
	DRW
	RTS

label MULTIPLICATION_SUBROUTINE
	MOV r2 r0
	LDI r0 #0
	CMP r1 r2
	MOV r3 r1 FLAG_LESS_THAN
	MOV r1 r2 FLAG_LESS_THAN
	MOV r2 r3 FLAG_LESS_THAN
label MULTIPLICATION_LOOP
	CMP r2 r2
	RTS FLAG_ZERO
	ADD r0 r1
	ADI r2 #65535
	JMP .MULTIPLICATION_LOOP

	var head_x
	var head_y
	var snake_color
